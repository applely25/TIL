# 생성자 함수

```jsx
function User(name, age){ //함수 이름 첫 글자는 대문자로 많이 씀
	//this = {}
	this.name = name; 
	this.age = age;
	this.sayName = function(){
			console.log(this.name); //this는 .앞에 것
	}
	//return this;
}
let user1 = new User('Mike', 30); //new연산자를 사용하여 호출
let user2 = new User('Jane', 22);
let user3 = User('Tom', 17); // undefined
//new 함수명{}
user1.sayName(); //'Mike'
```
